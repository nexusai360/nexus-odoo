## Queries de validação

```sql
-- Passo 2/9: recusas auditadas (não deve incrementar llm_usage)
SELECT action, count(*) FROM audit_logs
  WHERE action='agent_permission_denied' GROUP BY 1;
SELECT id, user_id, details->>'questionSnippet' AS pergunta,
       details->'deniedDomains' AS negados, created_at
  FROM audit_logs WHERE action='agent_permission_denied'
  ORDER BY created_at DESC LIMIT 5;

-- Passo 2: llm_usage NÃO deve crescer entre antes/depois da recusa
SELECT id, created_at, tokens_input, tokens_output, cost_usd
  FROM llm_usage ORDER BY created_at DESC LIMIT 5;

-- Passo 10: desfecho da decisão do router
SELECT id, user_question, picked_domains, outcome, created_at
  FROM agent_router_decision WHERE outcome='permission_denied'
  ORDER BY created_at DESC LIMIT 5;

-- Passo 11: nenhum domínio legado concedido
SELECT count(*) FROM user_domain_access WHERE domain IN ('rh','producao');  -- = 0
```
