```ts
// condicoes SQL reusaveis (item tem entrada_saida/situacao_nfe/data_emissao/empresa_id desnormalizados)
const cond: Prisma.Sql[] = [Prisma.sql`i.entrada_saida = '1'`, Prisma.sql`i.situacao_nfe = 'autorizada'`];
if (input.periodoDe && input.periodoAte) cond.push(Prisma.sql`i.data_emissao >= ${`${input.periodoDe}T00:00:00Z`}::timestamptz AND i.data_emissao < (${`${input.periodoAte}T00:00:00Z`}::timestamptz + interval '1 day')`);
if (input.empresaId !== undefined) cond.push(Prisma.sql`i.empresa_id = ${input.empresaId}`);
const condSql = Prisma.join(cond, " AND ");

const semCfopFinRows = await prisma.$queryRaw<Array<{ finalidade: string | null; n: bigint; v: Prisma.Decimal | null }>>(
  Prisma.sql`SELECT nf.finalidade_nfe AS finalidade, COUNT(*) n, COALESCE(SUM(i.vr_produtos),0) v
             FROM fato_nota_fiscal_item i JOIN fato_nota_fiscal nf ON nf.odoo_id = i.documento_id
             WHERE ${condSql} AND i.cfop_id IS NULL GROUP BY nf.finalidade_nfe ORDER BY v DESC`,
);
const semCfopPorFinalidade = semCfopFinRows.map((r) => ({
  finalidade: r.finalidade ?? "(sem)", totalItens: Number(r.n), valorProdutos: Number(r.v ?? 0),
}));

const outrasRows = await prisma.$queryRaw<Array<{ finalidade: string | null; v: Prisma.Decimal | null; n: bigint }>>(
  Prisma.sql`SELECT nf.finalidade_nfe AS finalidade, COUNT(*) n, COALESCE(SUM(i.vr_produtos),0) v
             FROM fato_nota_fiscal_item i JOIN fato_nota_fiscal nf ON nf.odoo_id = i.documento_id
             WHERE ${condSql} AND (i.cfop_nome LIKE '5949%' OR i.cfop_nome LIKE '6949%') GROUP BY nf.finalidade_nfe`,
);
const outrasNaoEspecificadas = {
  totalItens: outrasRows.reduce((s, r) => s + Number(r.n), 0),
  valorProdutos: outrasRows.reduce((s, r) => s + Number(r.v ?? 0), 0),
  valorFinalidadeVenda: outrasRows.filter((r) => r.finalidade === "1").reduce((s, r) => s + Number(r.v ?? 0), 0),
};
```
  Adicionar ao tipo `FaturamentoPorCfopResultado` (aditivo) e ao `return`. **Não tocar** `semCfop`.
